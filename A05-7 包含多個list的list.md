# A05-7 包含多個list的list


## (1) 程式範例
``` python
#---------------------------------------------
# 印出list的資料, list元素內容:
# 序號(0), 性別(1), 國文(2), 英文(3), 數學(4)
#---------------------------------------------

# 一個包括多個list的list
items = [[1,'男',46,56,42], [2,'男',48,51,39], [3,'男',50,52,47], [4,'女',40,56,45],
        [5,'男',43,41,29], [6,'女',44,52,50], [7,'女',44,56,47], [8,'男',50,56,47],
        [9,'男',56,60,30], [10,'男',43,55,49]]

# 逐行印出list的資料
for item in items:
    print(item)
```


## (4) 程式範例
``` python
#------------------------------------------------
# 印出list的資料, 依(1)總分, (2)國文由大到小排序
# 序號(0), 性別(1), 國文(2), 英文(3), 數學(4)
#------------------------------------------------

# 一個包括多個list的list
items = [[1,'男',46,56,42], [2,'男',48,51,39], [3,'男',50,52,47], [4,'女',40,56,45],
        [5,'男',43,41,29], [6,'女',44,52,50], [7,'女',44,56,47], [8,'男',50,56,47],
        [9,'男',56,60,30], [10,'男',43,55,49]]

# 逐行處理讀入的資料
for item in items:
    tot = item[2] + item[3] + item[4]
    item.append(tot)
    print(item)

# 分隔線 
print('-'*30)

# 依(1)總分, (2)國文排序
newItems = sorted(items, key=lambda x:(x[5], x[2]), reverse=True)

# 印出排序後結果
for item in newItems:    
    print(item)
```
